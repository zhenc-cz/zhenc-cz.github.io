var calPVI = function(image) {
  var allBands = image.select(['SR_B1', 'SR_B2', 'SR_B3', 'SR_B4', 'SR_B5', 'SR_B6', 'SR_B7']);
  var swirBands = image.select(['SR_B6', 'SR_B7']);
  var maxValue = allBands.reduce(ee.Reducer.max());
  var isSwirMax = swirBands.reduce(ee.Reducer.max()).eq(maxValue);
  var S = isSwirMax.where(isSwirMax, 1).where(isSwirMax.not(), -1);

  var kurtosis = image.reduce(ee.Reducer.kurtosis());
  var C = 1; // é˜ˆå€¼
  var PVI = S.multiply(C).add(kurtosis);

  return image.addBands(PVI.rename('PVI'));
};

};
